CSS Translator

This translator extracts comments from CSS (and also LESS and SCSS) files and splits the file along the comments. It recognizes only te JSDoc /** comments and leaves the other comments alone.

css-translator imports 
import * as bt from './base-translator'

export class CssTranslator extends bt.Translator {
    protected createBlocks() {
        let cssFile = this.outputFile.source as bt.OtherFile
        this.splitCss(cssFile.contents, cssFile.fileName)
    }

    protected splitCss(text: string, fileName: string) {
        this.scan(text, /\/\*\*(.*)\*\//gs,
            match => {
                this.openMarkdownBlock(match[1])
            },
            t => this.openCodeBlock(t))
    }
}
